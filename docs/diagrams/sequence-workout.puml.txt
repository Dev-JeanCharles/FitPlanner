  @startuml sequence-workout
  title Cliente Cria seu Treino - FitPlanner
  
  actor Cliente
  participant "API Gateway" as GW
  participant "fit-service" as FS
  database "PostgreSQL" as DB
  
  Cliente -> GW: POST /api/fit/splits\n{name, description}
  GW -> FS: forward (JWT validated)
  FS -> DB: INSERT workout_splits
  FS --> GW: 201 Created {split}
  GW --> Cliente: 201 Created
  
  Cliente -> GW: POST /api/fit/splits/{id}/days\n{day_of_week, muscle_group}
  GW -> FS: forward
  FS -> DB: INSERT workout_days
  FS --> GW: 201 Created {day}
  GW --> Cliente: 201 Created
  
  Cliente -> GW: POST /api/fit/days/{id}/exercises\n{name, sets, reps, rest_seconds}
  GW -> FS: forward
  FS -> DB: INSERT exercises
  FS --> GW: 201 Created {exercise}
  GW --> Cliente: 201 Created
  
  == Alteração de exercício ==
  
  Cliente -> GW: PUT /api/fit/exercises/{id}\n{name, sets, reps, change_reason}
  GW -> FS: forward
  FS -> DB: INSERT exercise_history (snapshot anterior)
  FS -> DB: UPDATE exercises (novos valores)
  FS --> GW: 200 OK
  GW --> Cliente: 200 OK
  
  @enduml
