  @startuml sequence-nutrition
  title Cliente Registra Nutrição - FitPlanner
  
  actor Cliente
  participant "API Gateway" as GW
  participant "fit-service" as FS
  database "PostgreSQL" as DB
  
  Cliente -> GW: POST /api/fit/nutrition/meals\n{meal_type, description,\ncalories, protein, carbs, fat}
  GW -> FS: forward (JWT validated)
  FS -> DB: UPSERT daily_nutrition_logs (cria se não existe)
  FS -> DB: INSERT meals
  FS -> DB: UPDATE daily_nutrition_logs (recalcula totais)
  FS --> GW: 201 Created {meal}
  GW --> Cliente: 201 Created
  
  == Registrar água ==
  
  Cliente -> GW: PATCH /api/fit/nutrition/water\n{amount_ml: 300}
  GW -> FS: forward
  FS -> DB: UPDATE daily_nutrition_logs\nSET water_ml += 300
  FS --> GW: 200 OK {water_ml: 2100}
  GW --> Cliente: 200 OK
  
  @enduml
