 @startuml sequence-notification
  title Fluxo de Notificação - FitPlanner
  
  participant "fit-service" as FS
  participant "calendar-service" as CS
  queue "AWS SQS" as SQS
  participant "notification-service" as NS
  database "MongoDB" as DB
  
  == Meta próxima do prazo (job agendado) ==
  
  FS -> FS: Scheduled job verifica\nmetas com end_date próximo
  FS -> SQS: publish "goal.deadline.approaching"\n{user_id, goal_title, days_remaining}
  
  SQS -> NS: consume
  NS -> DB: GET notification_preferences
  alt notificação habilitada
    NS -> NS: Envia e-mail/push
    NS -> DB: INSERT notification {status: SENT}
  end
  
  == Evento de calendário criado ==
  
  CS -> SQS: publish "event.created"\n{user_id, title, start_date}
  
  SQS -> NS: consume
  NS -> DB: INSERT notification\n{title: "Lembrete: {event.title}"}
  
  @enduml
